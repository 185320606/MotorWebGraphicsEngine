让视角沿着一定路径移动，可以制作漫游动画

### 1. 基本操作
```javascript
//初始化路径漫游管理器
let autoRoamManger = new Motor.AutoRoamManager(viewer);//传入Motor.Viewer对象
//从文件中加载路径关键点
autoRoamManger.createRouteFromFile('./assets/data/routes2.json');
//从数组对象中读取路径关键点
autoRoamManger.createRouteFromArray(array);
//将当前相机位置和姿态添加到路径中
autoRoamManger.addPose();
//播放路径
autoRoamManger.playRoam();
//停止路径
autoRoamManger.stopRoam();
//清空路径
autoRoamManger.emptyRoutes();
//获取路径点，输出的文本就可以直接保存成路径文件
console.log(JSON.stringify(autoRoamManger.cameraPoses));
```

### 2. 录制动画
```javascript
//initAutoRoam和closeRecord是两个Button DOM元素
//开启录制, 每隔一秒添加一次相机姿态
initAutoRoam.onclick = function(){
  timer = setInterval(function(){
    autoRoamManger.addPose();
  },1000);
}
//关闭录制
closeRecord.onclick = function(){
  clearInterval(timer);
}
//播放录制下来的路径动画
autoRoamManger.playRoam();
```
